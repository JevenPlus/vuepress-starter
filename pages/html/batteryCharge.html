let precent = 0

const el = document.querySelector('.text')

let timer = null

clearInterval(timer)
timer = setInterval(() => {
  precent = +(precent + 0.1).toFixed(1)
  if (precent > 100) {
    clearInterval(timer)
    return
  }
  el.innerText = precent + '%'
}, 200)
